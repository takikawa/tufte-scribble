\renewcommand{\preDoc}{\frontmatter \Tblankpage}

% Inserts a blank page
\newcommand{\Tblankpage}{\newpage\hbox{}\thispagestyle{empty}\newpage}

% Prints an epigraph and speaker in sans serif, all-caps type.
\newcommand{\Topenepigraph}[2]{%
  %\sffamily\fontsize{14}{16}\selectfont
  \begin{fullwidth}
  \sffamily\large
  \begin{doublespace}
  \noindent\allcaps{#1}\\% epigraph
  \noindent\allcaps{#2}% author
  \end{doublespace}
  \end{fullwidth}
}

\newcommand{\Tepigraphs}[1]{%
  \newpage\thispagestyle{empty}
  #1
}

% copyright page from the sample book
\newcommand{\Tcopyrightpage}[1]{%
\begin{fullwidth}
~\vfill
\thispagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\parskip}{\baselineskip}
Copyright \copyright\ \the\year\ \thanklessauthor

#1
\par\textit{First printing, \Tmonthyear}
\end{fullwidth}
}

% Prints the month name (e.g., January) and the year (e.g., 2008)
\newcommand{\Tmonthyear}{%
  \ifcase\month\or January\or February\or March\or April\or May\or June\or
  July\or August\or September\or October\or November\or
  December\fi\space\number\year
}
